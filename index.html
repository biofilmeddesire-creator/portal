function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === 'getLandlordData') { return getLandlordData(data.token); } const result = processForm(data); return ContentService.createTextOutput(JSON.stringify({ status: result })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function getLandlordData(token) { try { const ss = getTargetSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); const headers = data[0]; const tokenIdx = getSafeCol(headers, "Security_Token") !== -1 ? getSafeCol(headers, "Security_Token") : getSafeCol(headers, "Location_Code"); const resNameIdx = getSafeCol(headers, "Res_Name"); const roomTypeIdx = getSafeCol(headers, "Room_Type"); const spotsIdx = getSafeCol(headers, "Availability_spots"); const liveStatusIdx = getSafeCol(headers, "Inventory_Live_Status") !== -1 ? getSafeCol(headers, "Inventory_Live_Status") : getSafeCol(headers, "Status"); const priceIdx = getSafeCol(headers, "Landlord_Base_Price") !== -1 ? getSafeCol(headers, "Landlord_Base_Price") : getSafeCol(headers, "Price_Monthly"); let resName = ""; let rooms = []; for (let i = 1; i < data.length; i++) { if (String(data[i][tokenIdx]).trim() === token && token !== "") { const currentLiveStatus = String(data[i][liveStatusIdx] || "").trim().toLowerCase(); if (currentLiveStatus !== "archived") { resName = data[i][resNameIdx] || resName; rooms.push({ rowIndex: i + 1, type: data[i][roomTypeIdx], price: data[i][priceIdx], spots: data[i][spotsIdx] }); } } } return ContentService.createTextOutput(JSON.stringify({ residenceName: resName || "Residence Update", rooms: rooms })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
